rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // 1) Mỗi shop CHỈ đọc/ghi dữ liệu của chính mình
    //    (shop này không thể xem dữ liệu shop kia)
    match /users/{uid}/{document=**} {
      allow read, write: if request.auth != null && request.auth.uid == uid;
    }

    // 2) Trạng thái gói / hết hạn:
    //    - Người dùng chỉ ĐỌC được dòng của mình
    //    - KHÔNG ai sửa được từ app (chỉ bạn sửa trong Firebase Console)
    match /accounts/{uid} {
      allow read:  if request.auth != null && request.auth.uid == uid;
      allow write: if false;
    }

    // 3) Link công nợ công khai (để khách xem mà không cần đăng nhập)
    match /shares/{token} {
      allow read: if true;
      allow create: if request.auth != null
                    && request.resource.data.ownerUid == request.auth.uid;
      allow update: if request.auth != null
                    && request.resource.data.ownerUid == request.auth.uid
                    && (!('ownerUid' in resource.data) || resource.data.ownerUid == request.auth.uid);
      allow delete: if request.auth != null
                    && (!('ownerUid' in resource.data) || resource.data.ownerUid == request.auth.uid);
    }
  }
}
